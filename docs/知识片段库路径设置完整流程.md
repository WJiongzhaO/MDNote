**文件**: `main.js`

```javascript
// 第490行: fragment:set-global-path IPC 处理器
ipcMain.handle('fragment:set-global-path', async (_event, customPath) => {
  // 1. 验证路径
  if (!customPath || !fs.existsSync(customPath)) {
    throw new Error('Invalid path or path does not exist');
  }
  
  // 2. 保存到配置文件
  const config = loadConfig();
  config.customGlobalDataPath = customPath;  // ⭐ 关键：保存到配置文件
  saveConfig(config);
  
  // 3. 更新运行时变量
  globalDataPath = customPath;  // ⭐ 更新全局变量
  
  // 4. 确保目录存在
  if (!fs.existsSync(globalDataPath)) {
    fs.mkdirSync(globalDataPath, { recursive: true });
  }
  
  return { success: true, path: customPath };
});
```

## 🔄 路径获取逻辑

### getGlobalDataPath() 函数 (main.js 第38行)

```javascript
function getGlobalDataPath() {
  const config = loadConfig();
  
  // 优先级1: 自定义全局路径（用户设置的）
  if (config.customGlobalDataPath && fs.existsSync(config.customGlobalDataPath)) {
    return config.customGlobalDataPath;  // ⭐ 用户设置的路径
  }
  
  // 优先级2: 默认路径
  if (!isDev) {
    // 生产环境: {appPath}/MDNoteGlobal
    return path.join(basePath, 'MDNoteGlobal');
  } else {
    // 开发环境: {userDataPath}/global
    return path.join(userDataPath, 'global');
  }
}
```

```
    ↓

    ↓
所有 fragment:* IPC 处理器使用
    ↓
知识片段数据读写
知识片段图片存储
知识片段图片路径解析
```

## ⚠️ 关键点

1. **路径存储位置**: `config.json` 中的 `customGlobalDataPath` 字段
2. **路径使用范围**: 所有 `fragment:*` IPC 处理器都使用 `getGlobalDataPath()`
3. **路径独立性**: 完全独立于项目数据路径（`customDataPath`）
4. **路径持久化**: 保存在配置文件中，应用重启后仍然有效
